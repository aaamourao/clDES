143 
144  StorageIndexSigned const n_marked = this->marked_states_.size();
145  StatesVector x{ static_cast<StorageIndexSigned>(this->states_number_),
146  n_marked };
147  x.reserve(this->marked_states_.size());
148 
149  {
150  auto pos = 0ul;

152  x.coeffRef(state, pos) = true;
153  ++pos;
154  }

156

184 template<uint8_t NEvents, typename StorageIndex>
185 typename DESystem<NEvents, StorageIndex>::StatesSet
186 DESystem<NEvents, StorageIndex>::TrimStates() const
187 {
188  auto accpartstl = AccessiblePart();
189  spp::sparse_hash_set<StorageIndex> accpart;
190  for (StorageIndex s : accpartstl) {
191  accpart.insert(s);
192  }
193 
194  auto const invgraph = bit_graph_.transpose();
195 
196  auto const n_marked = this->marked_states_.size();
197 
198  StatesVector x{ static_cast<StorageIndexSigned>(this->states_number_),
199  static_cast<StorageIndexSigned>(n_marked) };
200  x.reserve(this->marked_states_.size());
201  std::vector<BitTriplet> xtriplet;
202 
203  {
204  auto pos = 0l;
205  for (StorageIndex state : this->marked_states_) {
206  xtriplet.push_back(BitTriplet(state, pos, true));
207  ++pos;
208  }
209  }
210  x.setFromTriplets(xtriplet.begin(),
211  xtriplet.end(),
212  [](bool const&, bool const&) { return true; });
213 
214  StatesVector y{ static_cast<StorageIndexSigned>(this->states_number_),
215  static_cast<StorageIndexSigned>(n_marked) };
216  auto n_accessed_states = 0l;
217  for (auto i = 0ul; i < this->states_number_; ++i) {
218  y = invgraph * x;
219 
220  if (n_accessed_states == y.nonZeros()) {
221  break;
222  } else {
223  n_accessed_states = y.nonZeros();
224  }
225 
226  x = y;
227  }
228 
229  y.pruned();
230 
231  StatesSet trimstates;
232  for (auto s = 0l; s < y.outerSize(); ++s) {
233  for (ColIteratorConst e(y, s); e; ++e) {
234  if (accpart.find(e.row()) != accpart.end()) {
235  trimstates.emplace(e.row());
236  }
237  }
238  }
239 
240  return trimstates;

243 template<uint8_t NEvents, typename StorageIndex>
244 DESystemBase<NEvents, StorageIndex>&
245 DESystem<NEvents, StorageIndex>::Trim()
246 {
247  auto trimstates = this->TrimStates();
248 
249  if (trimstates.size() == static_cast<size_t>(graph_.rows())) {
250  return *this;
251  }
252 
253  // States map: old state pos -> new state pos
254  std::vector<StorageIndexSigned> statesmap(this->states_number_, -1);
255 
256  // Copy graph and resize it
257  auto const old_graph = graph_;
258  this->states_number_ = trimstates.size();
259  graph_.resize(static_cast<StorageIndexSigned>(this->states_number_),
260  static_cast<StorageIndexSigned>(this->states_number_));
261  bit_graph_.resize(static_cast<StorageIndexSigned>(this->states_number_),
262  static_cast<StorageIndexSigned>(this->states_number_));
263 
264  if (this->states_events_.size() > 0) {
265  this->states_events_.erase(this->states_events_.begin() +
266  this->states_number_,
267  this->states_events_.end());
268  this->inv_states_events_.erase(this->inv_states_events_.begin() +
269  this->states_number_,
270  this->inv_states_events_.end());
271  }
272 
273  this->events_.reset();
274 
275  // Calculate the sparsity pattern
276  auto sparcitypattern = this->events_.count() * this->states_number_;
277 
278  // Fill statesmap
279  {
280  auto d = 0ul;
281  for (StorageIndex s : trimstates) {
282  statesmap[s] = d;
283  ++d;
284  }
285  }
286 
287  std::vector<Triplet> triplet;
288  std::vector<BitTriplet> bittriplet;
289 
290  triplet.reserve(sparcitypattern);
291  bittriplet.reserve(sparcitypattern);
292 
293  // Build new graph_ slice by slice
294  {
295  auto row_id = 0ul;
296  for (StorageIndex s : trimstates) {
297  if (this->states_events_.size() > 0) {
298  this->states_events_[row_id].reset();
299  this->inv_states_events_[row_id].reset();
300  }
301  for (RowIteratorGraph e(old_graph, s); e; ++e) {
302  if (statesmap[e.col()] != -1) {
303  auto const col_id = statesmap[e.col()];
304 
305  triplet.push_back(Triplet(row_id, col_id, e.value()));
306  bittriplet.push_back(BitTriplet(col_id, row_id, true));
307  this->events_ |= e.value();
308  if (this->states_events_.size() > 0) {
309  this->states_events_[row_id] |= e.value();
310  this->inv_states_events_[col_id] |= e.value();
311  }
312  }
313  }
314  ++row_id;
315  }
316  }
317 
318  // Remove aditional space
319  graph_.setFromTriplets(triplet.begin(), triplet.end());
320  bit_graph_.setFromTriplets(bittriplet.begin(),
321  bittriplet.end(),
322  [](bool const&, bool const&) { return true; });
323 
324  graph_.makeCompressed();
325  bit_graph_.makeCompressed();
326 
327  // Calculate new marked states
328  auto const old_marked = this->marked_states_;
329  this->marked_states_.clear();
330  for (StorageIndex s : old_marked) {
331  if (statesmap[s] != -1) {
332  this->marked_states_.emplace(statesmap[s]);
333  }
334  }
335 
336  return *this;
337 }

448 template<uint8_t NEvents, typename StorageIndex>
449 template<class StatesType>
450 std::shared_ptr<typename DESystem<NEvents, StorageIndex>::StatesSet>
451 DESystem<NEvents, StorageIndex>::Bfs_(
452  StatesType const& aInitialNodes,
453  std::function<void(StorageIndex const&, StorageIndex const&)> const&
454  aBfsVisit) const
455 {
456  /*
457  * BFS on a Linear Algebra approach:
458  * Y = G^T * X
459  */
460  // There is no need of search if a marked state is coaccessible
461  StatesVector host_x{ static_cast<StorageIndexSigned>(this->states_number_),
462  static_cast<StorageIndexSigned>(
463  aInitialNodes.size()) };
464 
465  // GPUs does not allow dynamic memory allocation. So, we have
466  // to set X on host first.
467  std::vector<StorageIndex> states_map;
468  for (auto state : aInitialNodes) {
469  host_x.coeffRef(state, states_map.size()) = true;
470  // Maping each search from each initial node to their correspondent
471  // vector on the matrix
472  states_map.push_back(state);
473  }
474 
475  return BfsCalc_(host_x, aBfsVisit, &states_map);
476 }

504 template<uint8_t NEvents, typename StorageIndex>
505 std::shared_ptr<typename DESystem<NEvents, StorageIndex>::StatesSet>
506 DESystem<NEvents, StorageIndex>::BfsCalc_(
507  StatesVector& aHostX,
508  std::function<void(StorageIndex const&, StorageIndex const&)> const&
509  aBfsVisit,
510  std::vector<StorageIndex> const* const aStatesMap) const
511 {
512  auto n_initial_nodes = aHostX.cols();
513 
518  StatesVector y{ static_cast<StorageIndexSigned>(this->states_number_),
519  static_cast<StorageIndexSigned>(n_initial_nodes) };
520  auto n_accessed_states = 0l;
521  for (StorageIndex i = 0ul; i < this->states_number_; ++i) {
522  y = bit_graph_ * aHostX;
523 
524  if (n_accessed_states == y.nonZeros()) {
525  break;
526  } else {
527  n_accessed_states = y.nonZeros();
528  }
529 
530  aHostX = y;
531  }
532 
534  if (aBfsVisit) {
535  for (auto s = 0l; s < y.rows(); ++s) {
536  for (ColIteratorConst e(y, s); e; ++e) {
537  aBfsVisit((*aStatesMap)[e.col()], e.row());
538  }
539  }
540  return nullptr;
541  }
542 
543  // Unfortunatelly, only C++17 allows shared_ptr to arrays
544  std::shared_ptr<StatesSet> accessed_states{
545  new StatesSet[n_initial_nodes], std::default_delete<StatesSet[]>()
546  };
547  for (auto s = 0l; s < y.cols(); ++s) {
548  for (ColIteratorConst e(y, s); e; ++e) {
549  accessed_states.get()[e.col()].emplace(e.row());
550  }
551  }
552  return accessed_states;
553 }
