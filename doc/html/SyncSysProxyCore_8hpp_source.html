40 template<uint8_t NEvents, typename StorageIndex>
41 op::SyncSysProxy<NEvents, StorageIndex>::SyncSysProxy(DESystemBase const& aSys0,
42  DESystemBase const& aSys1)
43  : DESystemBase{ aSys0.GetStatesNumber() * aSys1.GetStatesNumber(),
44  aSys1.GetInitialState() * aSys0.GetStatesNumber() +
45  aSys0.GetInitialState() }
46  , sys0_{ aSys0 }
47  , sys1_{ aSys1 }
48 {
49  sys_ptr_ = nullptr;
50  n_states_sys0_ = aSys0.GetStatesNumber();
51 
52  auto const in_both = aSys0.GetEvents() & aSys1.GetEvents();
53 
54  only_in_0_ = aSys0.GetEvents() ^ in_both;
55  only_in_1_ = aSys1.GetEvents() ^ in_both;
56 
57  // Initialize events_ from base class
58  this->events_ = aSys0.GetEvents() | aSys1.GetEvents();
59 
60  for (auto q0 : aSys0.GetMarkedStates()) {
61  for (auto q1 : aSys1.GetMarkedStates()) {
62  this->marked_states_.emplace(q1 * n_states_sys0_ + q0);
63  }
64  }
65 }

137  return false;

141  auto const qy = aQ / n_states_sys0_;

149  (in_y && only_in_1_.test(aEvent))) {
150  contains = true;
151  }
152 
153  return contains;

158 op::SyncSysProxy<NEvents, StorageIndex>::Trans(StorageIndex const& aQ,

172  (in_y && only_in_1_.test(aEvent)))) {
173  return -1;

175

177  auto const q0 = sys0_.Trans(qx, aEvent);
178  auto const q1 = sys1_.Trans(qy, aEvent);

180  return q1 * n_states_sys0_ + q0;

217 StatesArray<StorageIndex>
218 op::SyncSysProxy<NEvents, StorageIndex>::InvTrans(
219  StorageIndex const& aQ,
220  ScalarType const& aEvent) const
221 {
222  StatesArray<StorageIndex> inv_transitions;
223 
224  if (!this->events_.test(aEvent)) {
225  return inv_transitions;
226  }
227  // q = (qx, qy)
228  auto const qx = aQ % n_states_sys0_;
229  auto const qy = aQ / n_states_sys0_;
230 
231  auto const in_x = sys0_.ContainsInvTrans(qx, aEvent);
232  auto const in_y = sys1_.ContainsInvTrans(qy, aEvent);
233 
234  if (!((in_x && in_y) || (in_x && only_in_0_.test(aEvent)) ||
235  (in_y && only_in_1_.test(aEvent)))) {
236  return inv_transitions;
237  }
238 
239  if (in_x && in_y) {
240  auto const inv_trans_0 = sys0_.InvTrans(qx, aEvent);
241  auto const inv_trans_1 = sys1_.InvTrans(qy, aEvent);
242 
243  inv_transitions.reserve(inv_trans_0.size() + inv_trans_1.size());
244 
245  for (auto q0 : inv_trans_0) {
246  for (auto q1 : inv_trans_1) {
247  auto const q_from = q1 * n_states_sys0_ + q0;
248  inv_transitions.push_back(q_from);
249  }
250  }
251  } else if (in_x) {
252  auto const inv_trans_0 = sys0_.InvTrans(qx, aEvent);
253 
254  inv_transitions.reserve(inv_trans_0.size());
255 
256  for (auto q : inv_trans_0) {
257  auto const q_from = qy * n_states_sys0_ + q;
258  inv_transitions.push_back(q_from);
259  }
260  } else { // in_y
261  auto const inv_trans_1 = sys1_.InvTrans(qy, aEvent);
262 
263  inv_transitions.reserve(inv_trans_1.size());
264 
265  for (auto q : inv_trans_1) {

267  inv_transitions.push_back(q_from);
